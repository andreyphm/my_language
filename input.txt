func main()
{
    var x = 0;
    var y = 0;

    while (x < 3)
    {
        x = x + 1;
        y = 0;

        while (y < 5)
        {
            y = y + 1;

            if (y == 2)
            {
                break;
            }
        }
    }

    return x * 10 + y;
}
